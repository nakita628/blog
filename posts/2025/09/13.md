# Zod

- [Zod](https://zod.dev/)

## Basic

:::code-group
```ts [index.ts]
import * as z from 'zod'

const SampleSchema = z.object({
  a: z.string(),
  b: z.string(),
  c: z.string(),
})

const valid = SampleSchema.safeParse({
  a: '1',
  b: '2',
  c: '3',
})

// { success: true, data: { a: '1', b: '2', c: '3' } }
console.log(valid)
```
:::
