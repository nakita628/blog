apps ​

 tsconfig.base.jsonをextendsするが、outdirはこちらで指定しないと、ビルド先のパスがおかしくなる。

json
{
  "name": "apps",
  "devDependencies": {
    "@types/node": "^22.5.1",
    "typescript": "^5.5.4"
  }
}
ts
import { fizzBuzz } from 'fizzbuzz'

console.log('fizz Buzz Start')

fizzBuzz(100)

console.log('fizz Buzz End')
json
{
  "extends": "../tsconfig.base.json",
  "compilerOptions": {
    "outDir": "./dist"
  },
  "include": ["src/**/*.ts"]
}

packages/fizzbuzz ​

 tsconfig.base.jsonをextendsするが、outdirはこちらで指定しないと、ビルド先のパスがおかしくなる。

json
{
  "name": "fizzbuzz",
  "main": "dist/index.js",
  "types": "dist/index.d.ts",
  "devDependencies": {
    "@types/node": "^22.5.1",
    "typescript": "^5.5.4"
  }
}
ts
export function fizzBuzz(n: number): void {
  for (let i = 1; i <= n; i++) {
    console.log(i % 15 === 0 ? 'FizzBuzz' : i % 3 === 0 ? 'Fizz' : i % 5 === 0 ? 'Buzz' : i)
  }
}
json
{
  "extends": "../../tsconfig.base.json",
  "compilerOptions": {
    "outDir": "./dist"
  },
  "include": ["src/**/*.ts"]
}
