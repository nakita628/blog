# Npm Workspace

## Directory Structure

```
.
├── apps
│   └── src
│       ├── dist
│       │   ├── index.d.ts
│       │   └── index.js
│       ├── index.ts
│       ├── package.json
│       └── tsconfig.json
├── biome.json
├── Makefile
├── node_modules
│   └──  *
├── package.json
├── package-lock.json
└── packages
    ├── fizzbuzz
    │   ├── dist
    │   │   ├── index.d.ts
    │   │   └── index.js
    │   ├── index.ts
    │   ├── package.json
    │   └── tsconfig.json
    └── tsconfig
        ├── package.json
        └── packages
            └── tsconfig.json
```

## package.json
```json [package.json]
{
  "name": "npm-workspace-test",
  "private": true,
  "scripts": {
    "fmt": "npx @biomejs/biome format --write ./"
  },
  "workspaces": ["apps/*", "packages/*"],
  "devDependencies": {
    "@biomejs/biome": "^1.9.4",
    "tsx": "^4.19.1"
  }
}
```

## packages
:::code-group
```ts [packages/fizzbuzz/index.ts]
export const fizzBuzz = (number: number): string => {
  if (number % 15 === 0) return 'FizzBuzz'
  if (number % 3 === 0) return 'Fizz'
  if (number % 5 === 0) return 'Buzz'
  return number.toString()
}
```
:::
